Квиз: Оператори инкрементирања и декрементирања¶

Питање 1¶

Питање 2¶

int x = 2;
int y = x++;
printf("%d%d", x, y);

Питање 3¶

int x = 2;
int y = ++x;
printf("%d%d", x, y);

Питање 4¶

int x = 2;
x = x++ + ++x;

Питање 5¶

int x = 10, y = 10;
int z = x++ - --y;

Питање 6¶

int x = 10, y = 10;
int z = ++x * --y;

Питање 7¶

double x = 10.0, y = 5.0;
double z = x++ / y--;

Питање 8¶

int x = 10, y = 5;
int z = ++x + y--;

Питање 9¶

int x = 10, y = 10;
printf("%d%d%d%d", x++, ++x, y--, --y);
