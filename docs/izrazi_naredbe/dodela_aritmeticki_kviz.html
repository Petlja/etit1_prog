Питање 3¶

#include <stdio.h>

int main(void)
{
    double res = 6 + 5 / 4 + 3;
    printf("%.2lf", res);
    return 0;
}

Питање 5¶

#include <stdio.h>

int main(void)
{
    int a = 5, b = 4, c = 3;
    double res = (double)a * b / c;
    printf("%.2lf", res);
    return 0;
}
